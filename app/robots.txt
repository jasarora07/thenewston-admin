import { MetadataRoute } from 'next'
 
export default function robots(): MetadataRoute.Metadata {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: '/admin/', // Disallow your admin panels if you have them
    },
    sitemap: 'https://thenewston.com/sitemap.xml',
  }
}
